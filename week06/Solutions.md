## Question 1

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
	int number, i, sum_of_fact=0;
	scanf("%d", &number);
	while(1){
		int last_digit = number % 10;
		int factorial = 1;
		for(i = 2; i <= last_digit; i++){
			factorial *= i;
		}
		sum_of_fact += factorial;
		number /= 10;
		if(number == 0){
			break;
		}
	}
	printf("%d", sum_of_fact);
	return 0;
}
```

### Alternative

| Output                                            | Similarly                                          |

| A B C D E<br />B C D E<br />C D E<br />D E<br />E | 1 2 3 4 5<br />2 3 4 5<br />3 4 5 <br />4 5<br />5 |
| A B C<br />B C<br />C                             | 1 2 3 <br />2 3 <br />3                            |

| Row  | Output    |
| :--- | --------- |
| 1    | 1 2 3 4 5 |
| 2    | 2 3 4 5   |
| 3    | 3 4 5     |
| 4    | 4 5       |
| 5    | 5         |

So, we iterate rows with a loop variable **i** from 1 to 5. At each row, we iterate from row number to 5 *(from i to 5)*. 

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
	int number, i, j;
	scanf("%d", &number);
	for(i = 0; i < number; i++){
		for(j = i; j < number; j++){
			printf("%c", 'A'+j);
		}
		printf("\n");
	}
	return 0;
}
```
